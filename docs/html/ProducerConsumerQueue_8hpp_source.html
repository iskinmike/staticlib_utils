132  template<class ...Args>
133  bool write(Args&&... recordArgs) {
134  auto const currentWrite = writeIndex_.load(std::memory_order_relaxed);
135  auto nextRecord = currentWrite + 1;
136  if (nextRecord == size_) {
137  nextRecord = 0;
138  }
139  if (nextRecord != readIndex_.load(std::memory_order_acquire)) {
140  new (&records_[currentWrite]) T(std::forward<Args>(recordArgs)...);
141  writeIndex_.store(nextRecord, std::memory_order_release);
142  return true;
143  }
144 
145  // queue is full
146  return false;
147  }

155  bool read(T& record) {
156  auto const currentRead = readIndex_.load(std::memory_order_relaxed);
157  if (currentRead == writeIndex_.load(std::memory_order_acquire)) {
158  // queue is empty
159  return false;
160  }
161 
162  auto nextRecord = currentRead + 1;
163  if (nextRecord == size_) {
164  nextRecord = 0;
165  }
166  record = std::move(records_[currentRead]);
167  records_[currentRead].~T();
168  readIndex_.store(nextRecord, std::memory_order_release);
169  return true;
170  }

177  T* frontPtr() {
178  auto const currentRead = readIndex_.load(std::memory_order_relaxed);
179  if (currentRead == writeIndex_.load(std::memory_order_acquire)) {
180  // queue is empty
181  return nullptr;

183  return &records_[currentRead];
