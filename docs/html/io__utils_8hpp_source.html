171 template <typename Source, std::streamsize source_eof=-1>
172 class unbuffered_source : public unbuffered_streambuf_base {
173  Source& source;
174  bool exhausted = false;
175 
176 public:
182  unbuffered_source(Source& source) : source(source) { }
183 
189  virtual int uflow() override {
190  if (exhausted) {
191  return std::char_traits<char>::eof();
192  } else {
193  throw UtilsException(TRACEMSG("'uflow' operation is not supported by unbuffered streambuf"));
194  }
195  }
196 
197 protected:
205  virtual std::streamsize xsgetn(char* s, std::streamsize count) override {
206  if (!exhausted) {
207  auto res = source.read(s, count);
208  if(source_eof != res) {
209  return res;
210  } else {
211  exhausted = true;
212  return 0;
213  }
214  } else {
215  return 0;
216  }
217  }
218 
219 private:
223  unbuffered_source& operator=(const unbuffered_source&) = delete;
224 };
